cat << 'EOF' > /usr/bin/check-pids
#!/bin/bash

# Check if a CPU core number is provided
if [ -z "$1" ]; then
  echo "Usage: $0 <CPU_CORE_NUMBER>"
  exit 1
fi

# CPU core number to filter by
CPU_CORE=$1

# List processes running on the specified CPU core
ps -e -o pid,ppid,psr,comm | awk -v cpu="$CPU_CORE" '$3 == cpu'
EOF
chmod +x /usr/bin/check-pids
